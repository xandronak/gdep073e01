1//! Dithering and halftone strategies to map RGB->Spectra6.
2//! Feature-gated implementations, no_std by default; FS requires alloc.
3
4use crate::palette::{add_bias, map_rgb_to_spectra6_nearest, Spectra6};
5
6/// Strategy trait for per-pixel mapping with spatial/temporal context.
7pub trait DitherStrategy {
8    /// Map an sRGB triple at pixel (x,y) to Spectra6.
9    /// `x,y` are absolute framebuffer coords for matrix patterns.
10    fn map(&mut self, x: u32, y: u32, rgb: [u8; 3]) -> Spectra6;
11}
12
13/// Ordered Bayer 4x4: zero-alloc, fast.
14#[cfg(feature = "dither-bayer")]
15pub struct Bayer4x4;
16
17#[cfg(feature = "dither-bayer")]
18impl DitherStrategy for Bayer4x4 {
19    fn map(&mut self, x: u32, y: u32, rgb: [u8; 3]) -> Spectra6 {
20        // 4x4 Bayer thresholds 0..15
21        // Source: standard Bayer matrix
22        const M: [[i16; 4]; 4] = [[0, 8, 2, 10], [12, 4, 14, 6], [3, 11, 1, 9], [15, 7, 13, 5]];
23        let t = M[(y as usize) & 3][(x as usize) & 3] as i16; // 0..15
24                                                              // Convert t to a small bias in -8..+7
25        let bias = t - 8;
26        // Apply slight luminance-ish bias equally to channels
27        let b = [bias, bias, bias];
28        let nudged = add_bias(rgb, b);
29        map_rgb_to_spectra6_nearest(nudged)
30    }
31}

33/// Floydâ€“Steinberg: keeps 2 lines of error (alloc).
34#[cfg(feature = "dither-fs")]
35pub struct FloydSteinberg {
36    width: u32,
37    /// Two rows of error, interleaved RGB, i16 range to hold accumulated error.
38    cur: alloc::vec::Vec<i16>,
39    nxt: alloc::vec::Vec<i16>,
40    x: u32,
41    y: u32,
42}
43
44#[cfg(feature = "dither-fs")]
45impl FloydSteinberg {
46    pub fn new(width: u32) -> Self {
47        let len = (width as usize) * 3;
48        Self {
49            width,
50            cur: alloc::vec![0; len],
51            nxt: alloc::vec![0; len],
52            x: 0,
53            y: 0,
54        }
55    }
56    /// Call at the start of each new scanline y to advance the buffers if needed.
57    pub fn start_line(&mut self, y: u32) {
58        if y != self.y {
59            core::mem::swap(&mut self.cur, &mut self.nxt);
60            for v in &mut self.nxt {
61                *v = 0;
62            }
63            self.y = y;
64            self.x = 0;
65        }
66    }
67}
68
69#[cfg(feature = "dither-fs")]
70impl DitherStrategy for FloydSteinberg {
71    fn map(&mut self, x: u32, y: u32, rgb: [u8; 3]) -> Spectra6 {
72        // Assume left-to-right scanline order. If new line, roll buffers.
73        if y != self.y || (x == 0 && self.x != 0) {
74            self.start_line(y);
75        }
76        self.x = x;
77        let idx = (x as usize) * 3;
78        let adj = [
79            crate::palette::clamp_u8(rgb[0] as i32 + self.cur[idx + 0] as i32),
80            crate::palette::clamp_u8(rgb[1] as i32 + self.cur[idx + 1] as i32),
81            crate::palette::clamp_u8(rgb[2] as i32 + self.cur[idx + 2] as i32),
82        ];
83        let q = map_rgb_to_spectra6_nearest(adj);
84        // Quantization error e = adj - q_color
85        let qc = match q {
86            Spectra6::White => [255, 255, 255],
87            Spectra6::Black => [0, 0, 0],
88            Spectra6::Yellow => [255, 255, 0],
89            Spectra6::Red => [255, 0, 0],
90            Spectra6::Green => [0, 255, 0],
91            Spectra6::Blue => [0, 0, 255],
92        };
93        let er = adj[0] as i16 - qc[0] as i16;
94        let eg = adj[1] as i16 - qc[1] as i16;
95        let eb = adj[2] as i16 - qc[2] as i16;
96        // Distribute error: right (7/16), down-left (3/16), down (5/16), down-right (1/16)
97        // Right neighbor
98        if x + 1 < self.width {
99            let j = idx + 3;
100            self.cur[j + 0] = self.cur[j + 0].saturating_add((er * 7) / 16);
101            self.cur[j + 1] = self.cur[j + 1].saturating_add((eg * 7) / 16);
102            self.cur[j + 2] = self.cur[j + 2].saturating_add((eb * 7) / 16);
103        }
104        // Next row indices
105        let below_base = idx;
106        // Down-left
107        if x > 0 {
108            let j = below_base - 3;
109            self.nxt[j + 0] = self.nxt[j + 0].saturating_add((er * 3) / 16);
110            self.nxt[j + 1] = self.nxt[j + 1].saturating_add((eg * 3) / 16);
111            self.nxt[j + 2] = self.nxt[j + 2].saturating_add((eb * 3) / 16);
112        }
113        // Down
114        {
115            let j = below_base;
116            self.nxt[j + 0] = self.nxt[j + 0].saturating_add((er * 5) / 16);
117            self.nxt[j + 1] = self.nxt[j + 1].saturating_add((eg * 5) / 16);
118            self.nxt[j + 2] = self.nxt[j + 2].saturating_add((eb * 5) / 16);
119        }
120        // Down-right
121        if x + 1 < self.width {
122            let j = below_base + 3;
123            self.nxt[j + 0] = self.nxt[j + 0].saturating_add((er * 1) / 16);
124            self.nxt[j + 1] = self.nxt[j + 1].saturating_add((eg * 1) / 16);
125            self.nxt[j + 2] = self.nxt[j + 2].saturating_add((eb * 1) / 16);
126        }
127        q
128    }
129}

131/// Halftone tiles 2x2/3x3 with discrete fill levels between two palette colors.
132#[cfg(feature = "halftone")]
133pub struct Halftone {
134    /// Use 2 for 2x2 tiles or 3 for 3x3.
135    pub tile: u8,
136}
137
138#[cfg(feature = "halftone")]
139impl Halftone {
140    pub fn new(tile: u8) -> Self {
141        Self {
142            tile: if tile < 2 { 2 } else { tile.min(3) },
143        }
144    }
145    #[inline]
146    fn level_from_rgb(rgb: [u8; 3]) -> u8 {
147        // Simple luminance approximation 0..255
148        let y = (3 * rgb[0] as u16 + 6 * rgb[1] as u16 + 1 * rgb[2] as u16) / 10;
149        // Map to 0, 64,128,192,255 ~ 5 levels
150        if y < 32 {
151            0
152        } else if y < 96 {
153            1
154        } else if y < 160 {
155            2
156        } else if y < 224 {
157            3
158        } else {
159            4
160        }
161    }
162}
163
164#[cfg(feature = "halftone")]
165impl DitherStrategy for Halftone {
166    fn map(&mut self, x: u32, y: u32, rgb: [u8; 3]) -> Spectra6 {
167        let lvl = Self::level_from_rgb(rgb);
168        let n = self.tile as u32;
169        let xi = (x % n) as u8;
170        let yi = (y % n) as u8;
171        // Between Black and White by default; colorized blends future work.
172        // 2x2 ordering for levels 0..4
173        let on = if self.tile == 2 {
174            // 2x2 pattern order: [ (0,0), (1,1), (1,0), (0,1) ]
175            let rank = match (xi, yi) {
176                (0, 0) => 0,
177                (1, 1) => 1,
178                (1, 0) => 2,
179                _ => 3,
180            };
181            lvl > rank
182        } else {
183            // 3x3 pattern order by Bayer-like ranks 0..8
184            let rank = match (xi, yi) {
185                (1, 1) => 0,
186                (2, 0) | (0, 2) => 1,
187                (0, 1) | (1, 0) | (1, 2) | (2, 1) => 2,
188                (0, 0) | (2, 2) => 3,
189                _ => 4,
190            };
191            lvl > rank
192        };
193        if on {
194            Spectra6::White
195        } else {
196            Spectra6::Black
197        }
198    }
199}

205    #[cfg(feature = "dither-bayer")]
206    #[test]
207    fn bayer_deterministic() {
208        let mut b = Bayer4x4;

215    #[test]

217        let mut h = Halftone::new(2);
218        // Dark should be black, light should be white at same coords
219        let c1 = h.map(0, 0, [10, 10, 10]);
220        let c2 = h.map(0, 0, [240, 240, 240]);
221        assert!(matches!(c1, Spectra6::Black));
222        assert!(matches!(c2, Spectra6::White));
