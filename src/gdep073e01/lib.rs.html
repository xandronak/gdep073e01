1//! Embedded-graphics driver for the GDEP073E01 7-color e-paper display.
2//!
3//! This driver provides an interface to the Good Display GDEP073E01 e-paper display,
4//! implementing the `embedded-graphics` traits `DrawTarget` and `OriginDimensions`.
5//!
6//! ## Features
7//! - 7-color display support (Black, White, Yellow, Red, Orange, Blue, Green)
8//! - Full `embedded-graphics` integration with `DrawTarget` trait
9//! - Hardware abstraction layer (HAL) compatibility
10//! - Power management with sleep modes
11//! - Efficient internal buffering
12//!
13//! ## Usage
14//!
15//! ```
16//! use embedded_graphics::prelude::*;
17//! use embedded_graphics::primitives::{PrimitiveStyle, Rectangle};
18//! use gdep073e01::{Gdep073e01, Color};
19//! use core::convert::Infallible;
20//! # use embedded_hal::spi::SpiDevice;
21//! # use embedded_hal::digital::{OutputPin, InputPin};
22//! # use embedded_hal::delay::DelayNs;
23//! # struct MockSpi; struct MockPin; struct MockDelay;
24//! # impl embedded_hal::spi::ErrorType for MockSpi { type Error = Infallible; }
25//! # impl SpiDevice<u8> for MockSpi { fn transaction(&mut self, _: &mut [embedded_hal::spi::Operation<'_, u8>]) -> Result<(), Self::Error> { Ok(()) } }
26//! # impl embedded_hal::digital::ErrorType for MockPin { type Error = Infallible; }
27//! # impl OutputPin for MockPin { fn set_low(&mut self) -> Result<(), Self::Error> { Ok(()) } fn set_high(&mut self) -> Result<(), Self::Error> { Ok(()) } }
28//! # impl InputPin for MockPin { fn is_high(&mut self) -> Result<bool, Self::Error> { Ok(false) } fn is_low(&mut self) -> Result<bool, Self::Error> { Ok(true) } }
29//! # impl DelayNs for MockDelay { fn delay_ns(&mut self, _: u32) {} }
30//! # let spi = MockSpi; let cs = MockPin; let dc = MockPin; let rst = MockPin; let busy = MockPin; let delay = MockDelay;
31//!
32//! let mut display = Gdep073e01::new(spi, cs, dc, rst, busy, delay);
33//!
34//! display.init().expect("Failed to initialize display");
35//!
36//! let style = PrimitiveStyle::with_fill(Color::Red);
37//! Rectangle::new(Point::new(10, 10), Size::new(50, 50))
38//!     .into_styled(style)
39//!     .draw(&mut display)
40//!     .expect("Failed to draw rectangle");
41//!
42//! display.flush().expect("Failed to update display");
43//! display.sleep().expect("Failed to enter sleep mode");
44//! ```

107    /// White color (0x01)

110    /// Yellow color (0x02)
111    Yellow = 0x02,
112    /// Red color (0x03)
113    Red = 0x03,
114    /// Orange color (0x04)
115    Orange = 0x04,
116    /// Blue color (0x05)
117    Blue = 0x05,
118    /// Green color (0x06)
119    Green = 0x06,
120}

126/// GDEP073E01 display driver.
127///
128/// This driver manages communication with the GDEP073E01 7-color e-paper display
129/// via SPI and provides embedded-graphics compatibility.
130///
131/// # Type Parameters
132///
133/// - `SPI`: SPI device implementing `SpiDevice<u8>`
134/// - `CS`: Chip select pin (active low)
135/// - `DC`: Data/command pin (high for data, low for command)
136/// - `RST`: Reset pin (active low)
137/// - `BUSY`: Busy indicator pin (high when display is busy)
138/// - `DELAY`: Delay provider implementing `DelayNs`
139pub struct Gdep073e01<SPI, CS, DC, RST, BUSY, DELAY> {
140    spi: SPI,
141    cs: CS,
142    dc: DC,
143    rst: RST,
144    busy: BUSY,
145    delay: DELAY,
146    buffer: Box<[u8]>,
147    _phantom: PhantomData<Color>,
148}
149
150/// Error types for the GDEP073E01 driver.
151#[derive(Debug)]
152pub enum Error<SpiE, PinE> {
153    /// SPI communication error
154    Spi(SpiE),
155    /// GPIO pin operation error
156    Pin(PinE),
157    /// Timeout waiting for display ready
158    Timeout,
159}

161impl<SPI, CS, DC, RST, BUSY, DELAY, SpiE, PinE> Gdep073e01<SPI, CS, DC, RST, BUSY, DELAY>
162where
163    SPI: SpiDevice<u8, Error = SpiE>,
164    CS: OutputPin<Error = PinE>,
165    DC: OutputPin<Error = PinE>,
166    RST: OutputPin<Error = PinE>,
167    BUSY: InputPin<Error = PinE>,
168    DELAY: DelayNs,
169{
170    /// Creates a new GDEP073E01 driver instance.
171    ///
172    /// # Arguments
173    ///
174    /// * `spi` - SPI device for communication
175    /// * `cs` - Chip select pin (active low)
176    /// * `dc` - Data/command selection pin
177    /// * `rst` - Reset pin (active low)
178    /// * `busy` - Busy status pin
179    /// * `delay` - Delay provider
180    ///
181    /// # Returns
182    ///
183    /// A new driver instance with an initialized buffer.
184    pub fn new(spi: SPI, cs: CS, dc: DC, rst: RST, busy: BUSY, delay: DELAY) -> Self {
185        let buffer = vec![0x11; BUFFER_SIZE].into_boxed_slice(); // Default to white
186
187        Self {
188            spi,
189            cs,
190            dc,
191            rst,
192            busy,
193            delay,
194            buffer,
195            _phantom: PhantomData,
196        }
197    }
198
199    /// Initializes the display.
200    ///
201    /// Performs hardware reset and sends the initialization sequence required
202    /// for proper display operation. This must be called before any drawing operations.
203    ///
204    /// # Errors
205    ///
206    /// Returns `Error::Spi` for SPI communication failures, `Error::Pin` for GPIO
207    /// errors, or `Error::Timeout` if the display doesn't respond within the timeout period.
208    pub fn init(&mut self) -> Result<(), Error<SpiE, PinE>> {
209        self.reset()?;
210        self.send_init_sequence()?;
211        self.power_on()
212    }
213
214    /// Puts the display into deep sleep mode.
215    ///
216    /// This significantly reduces power consumption. The display requires
217    /// reinitialization via `init()` to wake up from deep sleep.
218    ///
219    /// # Errors
220    ///
221    /// Returns errors for communication failures or timeout.
222    pub fn sleep(&mut self) -> Result<(), Error<SpiE, PinE>> {
223        self.power_off()?;
224        self.command_with_data(CMD_DEEP_SLEEP, &[0xA5])
225    }
226
227    /// Updates the display with the current buffer contents.
228    ///
229    /// Sends the internal buffer to the display and triggers a refresh.
230    /// This operation may take several seconds to complete.
231    ///
232    /// # Errors
233    ///
234    /// Returns errors for communication failures or timeout.
235    pub fn flush(&mut self) -> Result<(), Error<SpiE, PinE>> {
236        self.write_command(CMD_DATA_START_TRANSMISSION)?;
237        self.write_buffer_data()?;
238        self.refresh()
239    }
240
241    /// Clears the internal buffer with the specified color.
242    ///
243    /// Note: This only affects the internal buffer. Call `flush()` to update the display.
244    ///
245    /// # Arguments
246    ///
247    /// * `color` - The color to fill the buffer with
248    pub fn clear_buffer(&mut self, color: Color) {
249        let color_val = color as u8;
250        let packed_color = (color_val << 4) | color_val;
251        self.buffer.fill(packed_color);
252    }
253
254    /// Sets a pixel in the internal buffer.
255    ///
256    /// Note: This only affects the internal buffer. Call `flush()` to update the display.
257    ///
258    /// # Arguments
259    ///
260    /// * `x` - X coordinate (0 to WIDTH-1)
261    /// * `y` - Y coordinate (0 to HEIGHT-1)
262    /// * `color` - Pixel color
263    pub fn set_pixel(&mut self, x: u32, y: u32, color: Color) {
264        if x >= WIDTH || y >= HEIGHT {
265            return;
266        }
267
268        let index = (y * WIDTH + x) as usize / 2;
269        let color_val = color as u8;
270        let mut byte = self.buffer[index];
271
272        if x % 2 == 0 {
273            byte = (byte & 0x0F) | (color_val << 4);
274        } else {
275            byte = (byte & 0xF0) | color_val;
276        }
277
278        self.buffer[index] = byte;
279    }
280
281    fn reset(&mut self) -> Result<(), Error<SpiE, PinE>> {
282        self.rst.set_low().map_err(Error::Pin)?;
283        self.delay.delay_ms(RESET_DELAY_MS);
284        self.rst.set_high().map_err(Error::Pin)?;
285        self.delay.delay_ms(RESET_DELAY_MS);
286        Ok(())
287    }
288
289    fn send_init_sequence(&mut self) -> Result<(), Error<SpiE, PinE>> {
290        self.command_with_data(CMD_CMDH, &[0x49, 0x55, 0x20, 0x08, 0x09, 0x18])?;
291        self.command_with_data(CMD_POWER_SETTING, &[0x3F])?;
292        self.command_with_data(CMD_PANEL_SETTING, &[0x5F, 0x69])?;
293        self.command_with_data(CMD_POFS, &[0x00, 0x54, 0x00, 0x44])?;
294        self.command_with_data(CMD_BOOSTER_SOFT_START1, &[0x40, 0x1F, 0x1F, 0x2C])?;
295        self.command_with_data(CMD_BOOSTER_SOFT_START2, &[0x6F, 0x1F, 0x17, 0x49])?;
296        self.command_with_data(CMD_BOOSTER_SOFT_START3, &[0x6F, 0x1F, 0x1F, 0x22])?;
297        self.command_with_data(CMD_PLL_CONTROL, &[0x08])?;
298        self.command_with_data(CMD_CDI, &[0x3F])?;
299        self.command_with_data(CMD_TCON_SETTING, &[0x02, 0x00])?;
300        self.command_with_data(CMD_TRES, &[0x03, 0x20, 0x01, 0xE0])?; // 800x480
301        self.command_with_data(CMD_T_VDCS, &[0x01])?;
302        self.command_with_data(CMD_PWS, &[0x2F])
303    }
304
305    fn write_command(&mut self, command: u8) -> Result<(), Error<SpiE, PinE>> {
306        self.dc.set_low().map_err(Error::Pin)?;
307        self.cs.set_low().map_err(Error::Pin)?;
308        let result = self.spi.write(&[command]).map_err(Error::Spi);
309        self.cs.set_high().map_err(Error::Pin)?;
310        result
311    }
312
313    fn write_data(&mut self, data: &[u8]) -> Result<(), Error<SpiE, PinE>> {
314        self.dc.set_high().map_err(Error::Pin)?;
315        self.cs.set_low().map_err(Error::Pin)?;
316        let result = self.spi.write(data).map_err(Error::Spi);
317        self.cs.set_high().map_err(Error::Pin)?;
318        result
319    }
320
321    fn command_with_data(&mut self, command: u8, data: &[u8]) -> Result<(), Error<SpiE, PinE>> {
322        self.write_command(command)?;
323        self.write_data(data)
324    }
325
326    fn write_buffer_data(&mut self) -> Result<(), Error<SpiE, PinE>> {
327        self.dc.set_high().map_err(Error::Pin)?;
328        self.cs.set_low().map_err(Error::Pin)?;
329
330        const CHUNK_SIZE: usize = 4096;
331        let mut result = Ok(());
332
333        for chunk in self.buffer.chunks(CHUNK_SIZE) {
334            if let Err(e) = self.spi.write(chunk).map_err(Error::Spi) {
335                result = Err(e);
336                break;
337            }
338        }
339
340        self.cs.set_high().map_err(Error::Pin)?;
341        result
342    }
343
344    fn wait_until_idle(&mut self) -> Result<(), Error<SpiE, PinE>> {
345        let mut remaining_delay = BUSY_TIMEOUT_MS;
346
347        while self.busy.is_high().map_err(Error::Pin)? {
348            if remaining_delay == 0 {
349                return Err(Error::Timeout);
350            }
351            let delay_step = remaining_delay.min(BUSY_WAIT_DELAY_MS);
352            self.delay.delay_ms(delay_step);
353            remaining_delay = remaining_delay.saturating_sub(delay_step);
354        }
355
356        Ok(())
357    }
358
359    fn power_on(&mut self) -> Result<(), Error<SpiE, PinE>> {
360        self.write_command(CMD_POWER_ON)?;
361        self.wait_until_idle()
362    }
363
364    fn power_off(&mut self) -> Result<(), Error<SpiE, PinE>> {
365        self.command_with_data(CMD_POWER_OFF, &[0x00])?;
366        self.wait_until_idle()
367    }
368
369    fn refresh(&mut self) -> Result<(), Error<SpiE, PinE>> {
370        self.command_with_data(CMD_DISPLAY_REFRESH, &[0x00])?;
371        self.wait_until_idle()
372    }
373}

375impl<SPI, CS, DC, RST, BUSY, DELAY, SpiE, PinE> DrawTarget
376    for Gdep073e01<SPI, CS, DC, RST, BUSY, DELAY>
377where
378    SPI: SpiDevice<u8, Error = SpiE>,
379    CS: OutputPin<Error = PinE>,
380    DC: OutputPin<Error = PinE>,
381    RST: OutputPin<Error = PinE>,
382    BUSY: InputPin<Error = PinE>,
383    DELAY: DelayNs,
384{
385    type Color = Color;
386    type Error = Error<SpiE, PinE>;
387
388    fn draw_iter<I>(&mut self, pixels: I) -> Result<(), Self::Error>
389    where
390        I: IntoIterator<Item = Pixel<Self::Color>>,
391    {
392        for Pixel(coord, color) in pixels {
393            if let Ok((x, y)) = coord.try_into() {
394                self.set_pixel(x, y, color);
395            }
396        }
397        Ok(())
398    }
399
400    fn fill_solid(&mut self, area: &Rectangle, color: Self::Color) -> Result<(), Self::Error> {
401        let area = area.intersection(&self.bounding_box());
402        if area.is_zero_sized() {
403            return Ok(());
404        }
405
406        let start_x = area.top_left.x as u32;
407        let start_y = area.top_left.y as u32;
408        let end_x = (area.top_left.x + area.size.width as i32) as u32;
409        let end_y = (area.top_left.y + area.size.height as i32) as u32;
410
411        for y in start_y..end_y {
412            for x in start_x..end_x {
413                self.set_pixel(x, y, color);
414            }
415        }
416
417        Ok(())
418    }
419
420    fn clear(&mut self, color: Self::Color) -> Result<(), Self::Error> {
421        self.clear_buffer(color);
422        Ok(())
423    }
424}
