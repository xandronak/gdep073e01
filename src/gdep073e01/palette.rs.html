1//! Spectra6 palette and mapping utilities.
2//! Works in no_std.

4/// Fixed Spectra 6 palette order used by the panelâ€™s LUT/driver (excluding Orange).
5#[derive(Copy, Clone, Debug, PartialEq, Eq)]
6pub enum Spectra6 {
7    White,
8    Black,
9    Yellow,
10    Red,
11    Green,
12    Blue,
13}
14
15/// Panel order palette RGB centers in sRGB 8-bit.
16/// Order: White, Black, Yellow, Red, Green, Blue
17pub const PALETTE: [[u8; 3]; 6] = [
18    [255, 255, 255], // White
19    [0, 0, 0],       // Black
20    [255, 255, 0],   // Yellow
21    [255, 0, 0],     // Red
22    [0, 255, 0],     // Green
23    [0, 0, 255],     // Blue
24];
25
26impl Spectra6 {
27    /// Convert this Spectra6 color to the driver's 7-color `Color` variant.
28    /// Note: Orange is intentionally not used by Spectra6.
29    pub fn to_driver_color(self) -> crate::Color {
30        match self {
31            Spectra6::White => crate::Color::White,
32            Spectra6::Black => crate::Color::Black,
33            Spectra6::Yellow => crate::Color::Yellow,
34            Spectra6::Red => crate::Color::Red,
35            Spectra6::Green => crate::Color::Green,
36            Spectra6::Blue => crate::Color::Blue,
37        }
38    }
39}

41/// Cheap perceptual-ish distance between two sRGB triples (0..=255).
42/// Uses a weighted squared distance to approximate luminance sensitivity without floats.
43#[inline]
44fn dist2_weighted(a: [u8; 3], b: [u8; 3]) -> u32 {
45    let dr = a[0] as i32 - b[0] as i32;
46    let dg = a[1] as i32 - b[1] as i32;
47    let db = a[2] as i32 - b[2] as i32;
48    // Weights approx human sensitivity: G strongest, then R, then B
49    // wR=3, wG=6, wB=1
50    (3 * dr * dr + 6 * dg * dg + 1 * db * db) as u32
51}
52
53/// RGB -> closest Spectra6 color (no dither).
54#[inline]
55pub fn map_rgb_to_spectra6_nearest(rgb: [u8; 3]) -> Spectra6 {
56    // Find minimum distance in PALETTE
57    let mut best = 0usize;
58    let mut best_d = u32::MAX;
59    for (i, p) in PALETTE.iter().enumerate() {
60        let d = dist2_weighted(rgb, *p);
61        if d < best_d {
62            best_d = d;
63            best = i;
64        }
65    }
66    match best {
67        0 => Spectra6::White,
68        1 => Spectra6::Black,
69        2 => Spectra6::Yellow,
70        3 => Spectra6::Red,
71        4 => Spectra6::Green,
72        _ => Spectra6::Blue,
73    }
74}

76/// Utility: clamp i32 to 0..=255 and return u8.
77#[inline]
78pub fn clamp_u8(v: i32) -> u8 {
79    if v < 0 {
80        0
81    } else if v > 255 {
82        255
83    } else {
84        v as u8
85    }
86}
87
88/// Utility: add bias to an rgb triple with saturation.
89#[inline]
90pub fn add_bias(rgb: [u8; 3], bias: [i16; 3]) -> [u8; 3] {
91    [
92        clamp_u8(rgb[0] as i32 + bias[0] as i32),
93        clamp_u8(rgb[1] as i32 + bias[1] as i32),
94        clamp_u8(rgb[2] as i32 + bias[2] as i32),
95    ]
96}

98#[cfg(test)]
99mod tests {
100    use super::*;
101
102    #[test]
103    fn nearest_basic() {
104        assert_eq!(
105            map_rgb_to_spectra6_nearest([250, 250, 250]),
106            Spectra6::White
107        );
108        assert_eq!(map_rgb_to_spectra6_nearest([5, 5, 5]), Spectra6::Black);
109        assert_eq!(
110            map_rgb_to_spectra6_nearest([250, 240, 10]),
111            Spectra6::Yellow
112        );
113        assert_eq!(map_rgb_to_spectra6_nearest([250, 10, 10]), Spectra6::Red);
114        assert_eq!(map_rgb_to_spectra6_nearest([10, 250, 10]), Spectra6::Green);
115        assert_eq!(map_rgb_to_spectra6_nearest([10, 10, 250]), Spectra6::Blue);
116    }
117
118    #[test]
119    fn spectra6_to_driver_color_nibbles() {
120        // Verify that Spectra6 maps to the native nibble codes used by the panel,
121        // as per the C++ reference _convert_to_native mapping.
122        assert_eq!(Spectra6::Black.to_driver_color() as u8, 0x00);
123        assert_eq!(Spectra6::White.to_driver_color() as u8, 0x01);
124        assert_eq!(Spectra6::Yellow.to_driver_color() as u8, 0x02);
125        assert_eq!(Spectra6::Red.to_driver_color() as u8, 0x03);
126        assert_eq!(Spectra6::Blue.to_driver_color() as u8, 0x05);
127        assert_eq!(Spectra6::Green.to_driver_color() as u8, 0x06);
128    }
