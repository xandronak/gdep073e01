1//! DitherDrawTarget adapter: converts Rgb888 to panel Color using a strategy.
2
3use embedded_graphics::{pixelcolor::Rgb888, prelude::*, primitives::Rectangle};
4
5use crate::dither::DitherStrategy;
6use crate::palette::map_rgb_to_spectra6_nearest;
7
8/// Wrap an embedded-graphics DrawTarget to apply palette+dither at draw time.
9pub struct DitherDrawTarget<T, S> {
10    inner: T,
11    strat: S,
12}
13
14impl<T, S> DitherDrawTarget<T, S> {
15    pub fn new(inner: T, strat: S) -> Self {
16        Self { inner, strat }
17    }
18    pub fn into_inner(self) -> T {
19        self.inner
20    }
21    pub fn inner_mut(&mut self) -> &mut T {
22        &mut self.inner
23    }
24    pub fn strategy_mut(&mut self) -> &mut S {
25        &mut self.strat
26    }
27}
28
29impl<T, S, E> DrawTarget for DitherDrawTarget<T, S>
30where
31    T: DrawTarget<Color = crate::Color, Error = E> + OriginDimensions,
32    S: DitherStrategy,
33{
34    type Color = Rgb888;
35    type Error = E;
36
37    fn draw_iter<I>(&mut self, pixels: I) -> Result<(), Self::Error>
38    where
39        I: IntoIterator<Item = embedded_graphics::Pixel<Self::Color>>,
40    {
41        for Pixel(coord, rgb) in pixels.into_iter() {
42            let (x, y) = (coord.x as u32, coord.y as u32);
43            let c6 = self
44                .strat
45                .map(x, y, [rgb.r(), rgb.g(), rgb.b()])
46                .to_driver_color();
47            // Forward as single pixel
48            self.inner.draw_iter(core::iter::once(Pixel(coord, c6)))?;
49        }
50        Ok(())
51    }
52
53    fn fill_solid(&mut self, area: &Rectangle, color: Self::Color) -> Result<(), Self::Error> {
54        // Use a simple iterator over the area mapping each pixel.
55        let rb = area.bounding_box();
56        let tl = rb.top_left;
57        let w = rb.size.width as i32;
58        let h = rb.size.height as i32;
59        for y in tl.y..(tl.y + h) {
60            for x in tl.x..(tl.x + w) {
61                let sx = x as u32;
62                let sy = y as u32;
63                let c6 = self
64                    .strat
65                    .map(sx, sy, [color.r(), color.g(), color.b()])
66                    .to_driver_color();
67                self.inner
68                    .draw_iter(core::iter::once(Pixel(Point::new(x, y), c6)))?;
69            }
70        }
71        Ok(())
72    }
73
74    fn clear(&mut self, color: Self::Color) -> Result<(), Self::Error> {
75        let c6 = map_rgb_to_spectra6_nearest([color.r(), color.g(), color.b()]).to_driver_color();
76        self.inner.clear(c6)
77    }
78}

80impl<T, S> OriginDimensions for DitherDrawTarget<T, S>
81where

84    fn size(&self) -> Size {
85        self.inner.size()
